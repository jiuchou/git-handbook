1.2.常用命令

1.2.常用命令

# 删除本地分支
git branch -d delete_branch_name

# 删除远程分支
git push origin :delete_branch_name
git push origin --delete delete_branch_name

# 添加仓库
git remote add origin http://github.com/xxx/xxx.git

# 查看
git status

# 添加
git add --all
git add .
git add file

# 撤销add操作
git reset HEAD
git reset HEAD file
